完整示例

// 定义一个处理函数
func 处理数据(list numbers) string {
    intnum 正数 = 0
    intnum 负数 = 0

    // 遍历统计
    for intnum i = 0; i < len(numbers); i = i + 1 {
        if numbers[i] > 0 {
            正数 = 正数 + 1
        } elif numbers[i] < 0 {
            负数 = 负数 + 1
        }
    }

    // 返回分析结果
    if 正数 > 负数 {
        return "正向趋势"
    } elif 负数 > 正数 {
        return "负向趋势"
    } else {
        return "平衡状态"
    }
}

// 使用示例
list 测试数据 = [1, -2, 3, -4, 5]
string 结果 = 处理数据(测试数据)
print "分析结果:", 结果
